CLASS Doctor
    ATTRIBUTES:
        name: String
        specialization: String
        experienceYears: Integer
    
    METHODS:
        CONSTRUCTOR(name, specialization, experienceYears)
            SET this.name = name
            SET this.specialization = specialization
            SET this.experienceYears = experienceYears
        
        getName(): String
            RETURN name
        
        getSpecialization(): String
            RETURN specialization
        
        getExperienceYears(): Integer
            RETURN experienceYears
        
        toString(): String
            RETURN name + " - " + specialization + " (" + experienceYears + " years)"

CLASS DepartmentDirectory
    ATTRIBUTES:
        departmentMap: Map<String, List<Doctor>>
    
    METHODS:
        CONSTRUCTOR()
            SET this.departmentMap = NEW Map<String, List<Doctor>>()
            initializeDepartments()
        
        initializeDepartments()
            SET departmentMap["Cardiology"] = NEW List<Doctor>()
            SET departmentMap["Neurology"] = NEW List<Doctor>()
            SET departmentMap["Pediatrics"] = NEW List<Doctor>()
            SET departmentMap["Orthopedics"] = NEW List<Doctor>()
        
        addDoctor(department, doctor)
            IF departmentMap.containsKey(department) THEN
                ADD doctor TO departmentMap[department]
                RETURN true
            ELSE
                RETURN false
            END IF
        
        addDoctorToDepartment(department, name, specialization, experienceYears): Boolean
            doctor = NEW Doctor(name, specialization, experienceYears)
            RETURN addDoctor(department, doctor)
        
        getDoctorsByDepartment(department): List<Doctor>
            IF departmentMap.containsKey(department) THEN
                RETURN departmentMap[department]
            ELSE
                RETURN NEW List<Doctor>()
            END IF
        
        displayDoctorsByDepartment(department)
            doctors = getDoctorsByDepartment(department)
            
            IF doctors.isEmpty() THEN
                DISPLAY "No doctors found in " + department + " department"
                RETURN
            END IF
            
            DISPLAY "Doctors in " + department + " Department:"
            DISPLAY "====================================="
            FOR i = 0 TO doctors.size() - 1
                doctor = doctors[i]
                DISPLAY (i + 1) + ". " + doctor.toString()
            END FOR
        
        addDepartment(departmentName): Boolean
            IF departmentMap.containsKey(departmentName) THEN
                RETURN false
            ELSE
                SET departmentMap[departmentName] = NEW List<Doctor>()
                RETURN true
            END IF
        
        removeDoctor(department, doctorName): Boolean
            IF departmentMap.containsKey(department) THEN
                doctors = departmentMap[department]
                FOR i = 0 TO doctors.size() - 1
                    IF doctors[i].getName() == doctorName THEN
                        REMOVE doctors[i] FROM doctors
                        RETURN true
                    END IF
                END FOR
            END IF
            RETURN false
        
        getDoctorsByExperience(minExperience): List<Doctor>
            experiencedDoctors = NEW List<Doctor>()
            
            FOR EACH entry IN departmentMap.entrySet()
                doctors = entry.getValue()
                FOR EACH doctor IN doctors
                    IF doctor.getExperienceYears() >= minExperience THEN
                        ADD doctor TO experiencedDoctors
                    END IF
                END FOR
            END FOR
            
            RETURN experiencedDoctors
        
        getDepartmentStatistics()
            DISPLAY "Hospital Department Statistics:"
            DISPLAY "============================="
            
            FOR EACH entry IN departmentMap.entrySet()
                department = entry.getKey()
                doctors = entry.getValue()
                DISPLAY department + ": " + doctors.size() + " doctors"
            END FOR
        
        displayAllDepartments()
            DISPLAY "Available Departments:"
            DISPLAY "====================="
            FOR EACH department IN departmentMap.keySet()
                doctorCount = departmentMap[department].size()
                DISPLAY department + " (" + doctorCount + " doctors)"
            END FOR
        
        searchDoctorByName(doctorName): Doctor
            FOR EACH entry IN departmentMap.entrySet()
                doctors = entry.getValue()
                FOR EACH doctor IN doctors
                    IF doctor.getName() == doctorName THEN
                        RETURN doctor
                    END IF
                END FOR
            END FOR
            RETURN null
        
        getTotalDoctors(): Integer
            total = 0
            FOR EACH entry IN departmentMap.entrySet()
                SET total = total + entry.getValue().size()
            END FOR
            RETURN total
